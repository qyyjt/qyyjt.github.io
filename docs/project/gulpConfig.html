    # APP内的自动化构建工具——gulp的相关配置

    # 什么是gulp?

    gulp是一个自动化构建工具，主要用来设定程序自动处理静态资源的工作。简单的说，gulp就是用来打包项目的。

    官网：https://gulpjs.com/ (opens new window)

    中文官网：https://www.gulpjs.com.cn/docs/ (opens new window)

    # 如何去实现一个简单的gulp配置？

    1. 全局安装：安装gulp命令行工具

      npm install -g gulp

    2. 创建项目目录并进入，输入下面的命令

      npm init

    3. 配置package.json文件（可选，不进行配置的话，可以走默认）

    4. 当前项目文件夹下输入命令

      npm install gulp --save-dev

    5. 当前项目下创建gulpfile.js文件，作为该项目配置文件。

      //gulpfile.js

var gulp = require('gulp');
gulp.task('default',function(){
    console.log('hello world');
});

      以上我们就实现了一个简单的gulp配置项目了。

    # 基于企业预警通APP端的gulp配置文件代码

    注意：当前的配置文件不可直接使用，需要修改对应的

    <span style="color: red">SVN项目目录：基本地址为</span>https://192.168.100.188/svn/FC_APP/HTML/trunk<span style="color: red">后面记得加上根目录的项目路径</span>

    <span style="color:red">内网服务器地址：基础地址为</span><span style="color:blue">/home/html</span><span style="color:red">后面记得加上根目录的项目路径</span>

    <span style="color:red">打包目录：项目内webpack配置的输出文件夹，一般默认为'./dist'</span>

    <span style="color:red;">提交日志：需要符合代码提交规范，【bug#XXXX】 禅道bug标题 + 当前功能的描述文字，【story#XXXX】 禅道需求标题 + 当前功能的描述文字</span>

    const gulp = require('gulp'),
  minimist = require('minimist'),
  gulpSSH = require('gulp-ssh'),
  shell = require('shelljs');

// SVN项目目录
const projectSVN = 'https://192.168.100.188/svn/FC_APP/HTML/trunk/hf/companyF9/financialData',

  // 内网服务器目录
  serverPath = '/home/html/hf/companyF9/financialData',
  serverPath2 = '/app/OpenResty/nginx/html/appdev_html/hf/companyF9/financialData',
  // 打包目录，请勿与SVN项目目录重名
  bundlePath = './dist',
  // 提交日志
  commitMsg = '#241 F9下方财务';

gulp.task('build', cb => {
  shell.exec('yarn build');
  cb();
});

gulp.task('checkout', cb => {
  shell.exec(`svn checkout ${projectSVN}`);
  cb();
});

gulp.task('checkIn', cb => {
  const msgOptions = {
    string: 'm',
    default: {
      m: commitMsg
    },
  };
  const options = minimist(process.argv.slice(2), msgOptions);
  const svnProjectPath = projectSVN.split('/').pop();

  shell.cd(svnProjectPath);
  shell.rm('-r', '*');
  shell.cd('..');
  shell.cp('-R', bundlePath + '/*', svnProjectPath);
  shell.cp('-R', './*.png', svnProjectPath);
  shell.cp('-R', './*.png', bundlePath);
  shell.cp('-R', './*.svg', svnProjectPath);
  shell.cp('-R', './*.svg', bundlePath);
  shell.cd(svnProjectPath);
  shell.exec('svn add * --force');
  shell.exec(`svn commit -m "${options.m}"`);
  shell.cd('..');
  cb();
});

gulp.task('removeMap', cb => {
  shell.rm(`${bundlePath}/**/assets/**/*.map`);
  cb();
});
// devwifi
gulp.task('upload', cb => {
  const ssh = new gulpSSH({
    ignoreErrors: false,
    sshConfig: {
      host: '10.15.97.42',
      username: 'root',
      password: 'jsoproject',
    },
  });
  setTimeout(() => {
    gulp.src(`${bundlePath}/**`).pipe(ssh.dest(serverPath));
    cb();
  }, 1000);
});

// dev
gulp.task('upload2', cb => {
  const ssh = new gulpSSH({
    ignoreErrors: false,

gulp.task('trunk', gulp.series('build', 'removeMap', 'checkout', 'checkIn', 'upload', 'upload2'));
gulp.task('default', gulp.series('trunk'));
